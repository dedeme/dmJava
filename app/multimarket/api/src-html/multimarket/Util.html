004package multimarket;
005
006import java.util.Map;
007import es.dm.Std;
008import es.dm.Cgi;
009import es.dm.Js;
010
011/** Utilities class */
012public class Util {
013  static Cgi cgi = null;
014
015  /** Create application Cgi */
016  public static void mkCgi () {
017    cgi = new Cgi(3600);
018  }
019
020  /**
021   * Returns application Cgi.
022   * @return Application Cgi.
023   */
024  public static Cgi getCgi () {
025    return cgi;
026  }
027
028  /**
029   * Produces a 'ok' response.
030   * @param rp Response
031   * @return Response
032   */
033  public static String ok (Map<String, Js> rp) {
034    return cgi.ok(rp);
035  }
036
037  /**
038   * Produces an empty response.
039   * @return Response
040   */
041  public static String empty () {
042    return cgi.empty();
043  }
044
045  /**
046   * Reads 'key' from 'rq' or throw an IllegalArgumentException.
047   * @param rq Request
048   * @param key Key
049   * @return Value of key
050   */
051  public static Js js (Map<String, Js> rq, String key) {
052    Js r = rq.get(key);
053    if (r == null) {
054      throw new IllegalArgumentException("Key '" + key + "' is missing");
055    }
056    return r;
057  }
058
059  /**
060   * Returns data directory
061   * @return Data directory
062   */
063  public static String dataDir () {
064    return Std.fpath(Std.home(), "data");
065  }
066
067  /**
068   * Does a task synchronizedly.
069   * @param fn Task to do
070   */
071  synchronized public static void sync (Runnable fn) {
072    fn.run();
073  }
074
075  static public class Wrapper<T> {
076    public T value;
077  }
078}
